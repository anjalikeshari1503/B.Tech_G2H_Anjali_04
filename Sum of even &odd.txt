#include <stdio.h>
int main() 
{
    int i,n;
    int oddSum=0,evenSum=0;
    printf("enter the number \n");
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        if(i%2==0) {
        evenSum +=i ;
        }
        else{
        oddSum += i;
        }
    }
    printf("sum of even numbers=%d\n",evenSum);
    printf("sum of odd numbers=%d \n",oddSum);
return 0;
}
