#include<stdio.h>
int main()
{
    int a,b;
    char ch;
    printf("enter + for addition\n");
    printf("enter - for subtraction\n");
    printf("enter * for multiplication\n");
    printf("enter / for division\n");
    printf("enter %% for reminder\n");
    scanf(" %c",&ch);
    printf("enter the two numbers a,b\n");
    scanf("%d%d",&a,&b);
    switch(ch)
    {
        case'+' :printf("addition=%d\n",(a+b));
        break;
        case'-' :printf("subtraction=%d\n",(a-b));
        break;
        case'*' :printf("multiplication=%d\n",(a*b));
        break;
        case'/' :printf("division=%d\n",(a/b));
        break;
        case'%' :printf("reminder=%d\n",(a%b));
        break;
        default:("Invalid choice\n");
    }
    return 0;
}
