#include <stdio.h>

int main() {
    int n ,sum=0;
    float avg ;
    printf("enter no. of elements\n");
   scanf("%d",&n);
   int arr[n];
  //input array arr[]
   printf("enter elements of array\n");
   for(int i=0;i<n;i++){
       scanf("%d",&arr[i]);
       sum = sum+arr[i];
       avg = sum /n;
   }
   printf("sum of elements of array=%d\n",sum);
   printf("average of elements of array=%f\n",avg);
    return 0;
}
