#include <stdio.h>

int main() {
    int n;
    printf("enter no. of elements\n");
   scanf("%d",&n);
   int arr[n];
  
   printf("enter elements of array\n");
   for(int i=0;i<n;i++){
       scanf("%d",&arr[i]);
   }
   int brr[n];
   for(int i=0;i<n;i++){
       brr[i]=arr[i];
   }
   printf("copied array\n");
   for(int i=0;i<n;i++){
       printf("%d",brr[i]);
   }
    return 0;
}
