#include<stdio.h>
int main()
{
    char ch ;
    printf("Enter the character\n");
    scanf("%c",&ch);
    if(ch >= 'A'&&ch <= 'Z'){
        printf("The alphabet is capital\n");
    }
    else if(ch >= 'a'&&ch <= 'z'){
        printf("The alphabet is small\n");
    }    
    else {
        printf("The character is not a alphabet \n");
    }
    return 0;
}
