#include <stdio.h>
#include <math.h>
int main()
{
    float p,r,t,ci,si,a ;
    printf("enter the value of p,r,t\n");
    scanf("%f%f%f",&p,&r,&t);
    si=p*r*t/100.0 ;
    printf("simple interest=%f\n",si);
    a=p*pow((1+r/100),t);
    printf("amount=%f\n",a);
    ci=a-p;
    printf("compound interest=%f\n",ci);
    return 0;
    }
