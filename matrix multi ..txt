#include <stdio.h>

int main() {
  int m,n,p;
  printf("enter no. of rows and columns\n");
  scanf("%d%d",&m,&n);
  printf("enter no. of column of matrix B\n");
  scanf("%d",&p);
  // input fisrt Matrix A
  int arr[m][n];
  printf("enter array elements of Matrix A\n");
  for(int i=0;i<m;i++){
      for(int j=0;j<n;j++){
          scanf("%d",&arr[i][j]);
      }
  }
  // input second Matrix B
  int brr[n][p];
  printf("enter array elements of Matrix B\n");
  for(int i=0;i<n;i++){
      for(int j=0;j<p;j++){
          scanf("%d",&brr[i][j]);
      }
  }
  // initilization of third Matrix C
  int crr[m][p];
  for(int i=0;i<m;i++){
      for(int j=0;j<p;j++){
         crr[i][j]=0 ;
      }
  }
  // Matrix multiplication 
  for(int i=0;i<m;i++){
      for(int j=0;j<p;j++){
          for(int k=0;k<n;k++)
           crr[i][j] +=arr[i][k]*brr[k][j];
      }
  }
  // print result 
  printf("resultant matrix\n");
  for(int i=0;i<m;i++){
      for(int j=0;j<p;j++){
          printf("%d",crr[i][j]);
      }
      printf("\n");
  }
    return 0;
}
