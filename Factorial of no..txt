#include <stdio.h>
int main() {
    int n, i;
    unsigned long long fact =1;
    printf("enter the value of n\n");
    scanf("%d",&n);
    if(n<0){
        printf("factorial is not defined for negative numbers\n");
    }
    else{
        for(i=1;i<=n;i++){
            fact=fact*i;
        }
        printf("factorial=%llu\n",fact);
    }
    return 0;
}
